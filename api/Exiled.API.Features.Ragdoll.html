    public static Ragdoll Spawn(RoleType roleType, DamageTypes.DamageType deathCause, string victimNick, Vector3 position, Quaternion rotation = default(Quaternion), Vector3 velocity = default(Vector3), bool allowRecall = false, int playerId = -1, bool scp096Death = false, string mirrorOwnerId = null)
    Parameters
    Type Name Description
    RoleType roleType

    The RoleType to use as ragdoll.
    DamageTypes.DamageType deathCause

    The death cause, expressed as a DamageTypes.DamageType.
    System.String victimNick

    The name from the victim, who the corpse belongs to.
    UnityEngine.Vector3 position

    Where the ragdoll will be spawned.
    UnityEngine.Quaternion rotation

    The rotation for the ragdoll.
    UnityEngine.Vector3 velocity

    The initial velocity the ragdoll will have, as if it was exploded.
    System.Boolean allowRecall

    Sets this ragdoll as respawnable by SCP-049. Must have a valid playerId.
    System.Int32 playerId

    Used for recall. The Id to be recalled.
    System.Boolean scp096Death

    Sets this ragdoll as Scp096's victim.
    System.String mirrorOwnerId

    Can be ignored. The Dissonance.Integrations.MirrorIgnorance.MirrorIgnorancePlayer's PlayerId field.
    Returns
    Type Description
    Ragdoll

    The spawned Ragdoll.
    Remarks

    Tip: You can do 'allowRecall: true, playerId: MyPlayer.Id' to skip parameters.

    Examples
    // Code to spawn a fake ragdoll
if (ev.Player == MyPlugin.TheInmortalPlayer)
{
    var fakeRagdoll = Ragdoll.Spawn(RoleType.ClassD, DamageTypes.Fall, "The Falling Guy", new Vector3(1234f, -1f, 4321f));
}
